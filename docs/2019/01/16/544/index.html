不敢私藏：Air202, 按键长按重启和特长按回退为出厂版本，感谢 李炜镪

module(..., package.seeall)

require 'pins'

local btnDown = false
local st = 0

pins.setup(pio.P0_2, function()
    if btnDown then
        local d = os.time() - st

        if d > 4 then
            log.info('BUTTON', 'RESTORE DEVICE...')
            assert(false)
        elseif d > 1 then
            log.INFO('BUTTON', 'RESTART DEVICE...')
            sys.restart('BUTTON')
        end

        st = 0
    else
        st = os.time()
    end
    btnDown = not btnDown
end)
